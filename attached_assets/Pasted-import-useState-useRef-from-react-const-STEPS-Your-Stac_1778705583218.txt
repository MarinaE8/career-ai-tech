// --- Sub-components ---
function Label({ children }) {
  return (
    <div style={{
      fontSize: 10, color: "#4a7060", letterSpacing: "0.12em",
      textTransform: "uppercase", marginBottom: 8, fontFamily: "monospace",
    }}>
      {children}
    </div>
  );
}

function Field({ label, value, onChange, placeholder, multiline, rows = 3 }) {
  const base = {
    width: "100%", background: "rgba(255,255,255,0.02)",
    border: "1px solid #1a1a1a", borderRadius: 6,
    padding: "11px 14px", color: "#c0d8cc", fontSize: 12,
    fontFamily: "'JetBrains Mono', monospace", resize: "vertical",
    lineHeight: 1.6, transition: "border-color 0.15s",
  };
  return (
    <div>
      <Label>{label}</Label>
      {multiline
        ? <textarea value={value} onChange={e => onChange(e.target.value)} placeholder={placeholder} rows={rows} style={base} />
        : <input value={value} onChange={e => onChange(e.target.value)} placeholder={placeholder} style={{ ...base, height: 42 }} />
      }
    </div>
  );
}

function CtaButton({ children, onClick, disabled, style = {} }) {
  return (
    <button onClick={onClick} disabled={disabled} style={{
      marginTop: 28, width: "100%", padding: "13px 20px",
      background: disabled ? "rgba(0,255,128,0.03)" : "rgba(0,255,128,0.1)",
      border: disabled ? "1px solid #1a1a1a" : "1px solid rgba(0,255,128,0.35)",
      borderRadius: 6, color: disabled ? "#333" : "#00ff80",
      fontFamily: "monospace", fontSize: 13, fontWeight: 700,
      cursor: disabled ? "not-allowed" : "pointer",
      letterSpacing: "0.05em", transition: "all 0.2s",
      ...style,
    }}>
      {children}
    </button>
  );
}
